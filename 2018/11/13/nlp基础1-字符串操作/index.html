字符串统计


import re 
from collections import Counter
def get_max_value(string):

    string = string.lower()
    result = re.findall(r'[a-z]',string)
    result = Counter(result)
    num_list = list(result.values())
    max_num = max(num_list)
    max_string =[]
    for k,v in result.items():
        if v == max_num:
            max_string.append(k)
    return sorted(max_string),max_num
print get_max_value('baifeng is learning NLP')

def get_max_value_2(string):
    w=Counter([i for i in string.lower() if i.isalpha()])
    # print type(w)
    max_num = max(w.values())
    result=[k for k,v in w.items() if v == max_num]
    return result
